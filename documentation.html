function getGitHubApiUrl(path) {
  if (GITHUB_PROXY_CONFIG.USE_PROXY) {
    return `${GITHUB_PROXY_CONFIG.PROXY_URL}/api/github/${path}`;
  }
  return `https://api.github.com/${path}`;
}

async function hashPassword(password) {
  const encoder = new TextEncoder();
  const data = encoder.encode(password);
  const hashBuffer = await crypto.subtle.digest('SHA-256', data);
  const hashArray = Array.from(new Uint8Array(hashBuffer));
  return hashArray.map(byte => 
    byte.toString(16).padStart(2, '0')

// Stored hash (generated offline)
const ADMIN_PASSWORD_HASH = 'abc123...';

// Verify login
async function verifyLogin(inputPassword) {
  const inputHash = await hashPassword(inputPassword);
  if (inputHash === ADMIN_PASSWORD_HASH) {
    sessionStorage.setItem('boss_auth', 'true');
    sessionStorage.setItem('boss_auth_time', Date.now());
    return true;
  }
  return false;
}

Security Features

  • Hash Comparison: Compares hashes, not plain passwords

7.1 CORS Issues

The Problem

GitHub API requests from frontend were blocked by CORS (Cross-Origin Resource Sharing) policy when using authentication tokens. Browser prevented requests due to security restrictions.

Error Message

Access to fetch at 'https://api.github.com/repos/...' from origin 
'https://rafid-003.github.io' has been blocked by CORS policy: 
Response to preflight request doesn't pass access control check.

The Solution

Implemented a Python Flask proxy server that handles all GitHub API requests server-side, avoiding CORS entirely.

Backend Proxy (secure-proxy-server.py)

from flask import Flask, jsonify, request
from flask_cors import CORS
import requests

app = Flask(__name__)
CORS(app, origins=['*'])  # Allow all origins

@app.route('/api/github/<path:github_path>')
def proxy_github(github_path):
    token = get_next_token()
    headers = {
        'Authorization': f'Bearer {token}',
        'Accept': 'application/vnd.github.v3+json'
    }
    
    response = requests.get(
        f'https://api.github.com/{github_path}',
        headers=headers,
        params=request.args
    )
    
    return jsonify(response.json()), response.status_code
